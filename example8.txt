is_string(X: 'string').
is_string(1)?
is_string(1.0)?
is_string("1")?

is_float(X: 'float').
is_float(1)?
is_float(1.0)?
is_float("1")?

is_integer(X: 'integer').
is_integer(1)?
is_integer(1.0)?
is_integer("1")?

is_numeric(X) :- is_float(X) | is_integer(X).
is_numeric(1)?
is_numeric(1.0)?
is_numeric("1")?
